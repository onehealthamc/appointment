function doPost(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let data; try { data = JSON.parse(e.postData.contents); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ result: 'error', message: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } if (data.type === 'reservation') { const sheet = ss.getSheetByName('예약 신청'); sheet.appendRow([ new Date().toLocaleString('ko-KR'), data.visitType || '', data.ownerName || '', data.phone || '', data.address || '', data.petName || '', data.petAge || '', data.petBirthdate || '', data.gender || '', data.breed || '', data.symptoms || '', data.extraPets || '', '미확인' ]); } if (data.type === 'change' || data.type === 'cancel') { const sheet = ss.getSheetByName('변경·취소'); sheet.appendRow([ new Date().toLocaleString('ko-KR'), data.type === 'change' ? '예약변경' : '예약취소', data.ownerName || '', data.phone || '', data.petName || '', data.apptDate || '', data.newDate || '', data.reason || '', '미처리' ]); } return ContentService .createTextOutput(JSON.stringify({ result: 'success' })) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { return ContentService .createTextOutput(JSON.stringify({ result: 'ok' })) .setMimeType(ContentService.MimeType.JSON); }
